POST

	"nikeName":"", --昵称, 租户下唯一
	"realName":"", --真实姓名
	"sex":1/2, -- 性别, 1男, 2女
	"phoneNum":"", -- 手机号码
	"email":"", -- 邮箱

	"loginPwd":"" -- 开通的账户的登录密码, 可不传, 默认为111111

{
